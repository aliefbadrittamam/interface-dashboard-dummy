Template Kode Integrasi IoT (Firebase/MQTT)

// firebase-integration.js // Copy ini ke file baru dan sesuaikan konfigurasi import { initializeApp } from "firebase/app"; import { getDatabase, ref, onValue, set } from "firebase/database"; // Konfigurasi Firebase (ganti dengan config project Anda) const firebaseConfig = { apiKey: "YOUR_API_KEY", databaseURL: "https://your-project.firebaseio.com", // ... config lainnya }; const app = initializeApp(firebaseConfig); const db = getDatabase(app); // Listener Node 1 (Pintu) onValue(ref(db, 'node1'), (snap) => { const data = snap.val(); state.node1_door.status = data.status; state.node1_door.rfidLog = data.rfidLog || []; renderNode1(); }); // Listener Node 2 (Gas & Motion) onValue(ref(db, 'node2'), (snap) => { const data = snap.val(); state.node2_safety.gasPPM = data.gasPPM; state.node2_safety.motionDetected = data.motionDetected; renderNode2(); }); // Listener Node 3 (Lampu & Alarm) onValue(ref(db, 'node3'), (snap) => { const data = snap.val(); state.node3_control.livingRoomLight = data.livingRoomLight; state.node3_control.porchLight = data.porchLight; state.node3_control.alarm.active = data.alarmActive; renderNode3(); }); // Kirim perintah ke hardware function sendCommand(node, action, value) { set(ref(db, `commands/${node}`), { action, value, timestamp: Date.now() }); } // Contoh: Toggle lampu dari dashboard document.getElementById('toggle-living").addEventListener("change", (e) => { sendCommand("node3", "toggleLivingRoom", e.target.checked); });
